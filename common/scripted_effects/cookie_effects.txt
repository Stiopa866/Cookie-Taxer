set_cookies = {
    [[value]
        set_variable = {
            which = temp1
            value = $value$
        }
        var_to_long = {
            DEC_VAR = temp1
            nameset = cookies
        }
    ]
    [[variable]
        var_to_long = {
            DEC_VAR = $variable$
            nameset = cookies
        }
    ]

    get_msd = { ln = cookies mantissa = cookies_mantissa exponent = cookies_exp }
}
