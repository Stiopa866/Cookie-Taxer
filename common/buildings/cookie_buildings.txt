ct_cursor = {
	cost = 0
	time = 1
	indestructible = yes
	onmap = no

	potential = {
		always = yes
	}

	build_trigger = {
		custom_trigger_tooltip = {
			tooltip = cursor_cost_tt
			FROM = {
				long_gte = { ln0 = cookies ln1 = cursor_cost }
			}
		}
		
	}

	modifier = { }

	on_built = { hidden_effect = {
		ROOT = { set_province_flag = has_cursor }
		FROM = { building_recount = yes }
	}}

	on_construction_started = { hidden_effect = {
		FROM = { subtract_longs = { ln1 = cookies ln2 = cursor_cost } }
		ROOT = { set_province_flag = has_cursor }
		FROM = { building_recount = yes }
	}}

	on_construction_canceled = { hidden_effect = {
		ROOT = { clr_province_flag = has_cursor }
		FROM = { building_recount = yes }
	}}

	on_destroyed = { hidden_effect = {
		ROOT = { clr_province_flag = has_cursor }
		FROM = { building_recount = yes }
	}}
}
